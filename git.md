# 1. Navigate to your project folder
cd "<your-base-path>\local-kong-hybrid"

# 2. Initialize a new Git repository
git init

# 3. Stage files for the first commit
git add .

# 4. Commit the initial repository
git commit -m "Initial commit: add .gitignore, folder structure, scripts, and configs"

# 5. Add remote repository (replace URL with your Git remote)
git remote add origin <your-repo-url>

# 6. Push to the remote repository
git branch -M main
git push -u origin main
